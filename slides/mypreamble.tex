% my takehome function
\newcommand{\takehome}[1]{%
  \bigskip
  \btVFill
  \pause
  \begin{tcolorbox}[colback=green!10!white, colframe=green!40!black,notitle,boxsep=0mm,
  fontupper=\sffamily\bfseries, halign=center, valign=center, natural height] 
  #1
  \end{tcolorbox}

}
